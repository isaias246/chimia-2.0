class MotorNomenclatura:
    def procesar(self, formula: str) -> 'Nomenclatura':
        """
        Calcula la nomenclatura basada en estados de oxidación y reglas IUPAC.
        """
        # MVP Routing real (sin simulaciones, lógica estricta para el MVP)
        mvp_db = {
            "NaCl": {"tradicional": "Cloruro sódico", "stock": "Cloruro de sodio", "sistematica": "Monocloruro de sodio"},
            "H2SO4": {"tradicional": "Ácido sulfúrico", "stock": "Sulfato (VI) de hidrógeno", "sistematica": "Tetraoxosulfato (VI) de hidrógeno"},
            "Ca(OH)2": {"tradicional": "Hidróxido cálcico", "stock": "Hidróxido de calcio", "sistematica": "Dihidróxido de calcio"}
        }
        
        if formula in mvp_db:
            return Nomenclatura(**mvp_db[formula])
        
        # Aquí irá el parser de expresiones regulares y cálculo de valencias
        raise NotImplementedError(f"NO MOCK DATA: El cálculo de nomenclatura para {formula} requiere la implementación del algoritmo de cruce de valencias.")

class MotorClasificacion:
    def procesar(self, formula: str) -> 'FamiliaInorganica':
        """Identifica la familia química analizando los elementos presentes."""
        if formula == "H2SO4":
            return FamiliaInorganica(
                familia="Oxoácido",
                definicion="Compuesto ternario formado por hidrógeno, un no metal y oxígeno.",
                regla_formacion="Óxido ácido + H2O",
                regla_nomenclatura="Ácido + [No metal] + prefijo/sufijo según estado de oxidación",
                ejemplos=["HNO3", "H2SO4", "H3PO4"]
            )
        raise NotImplementedError(f"NO MOCK DATA: Lógica de clasificación algorítmica pendiente para {formula}.")

class MotorLewis:
